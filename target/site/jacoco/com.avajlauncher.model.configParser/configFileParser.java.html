package com.avajlauncher.model.configParser;

import java.util.HashMap;
import java.util.Map;

import com.avajlauncher.model.coordinates.Coordinates;

public class configFileParser {
	private String line;
	private Map<String, String>	tokenizedLine;


	public void setLine(final String p_line) {
		this.line = p_line;
	}
	
	public void tokenizeLine() {
		// check error on line
		String[] splittedLine = this.line.split("\\s+");
		this.tokenizedLine = new HashMap<String, String>();

		this.tokenizedLine.put("type", splittedLine[0]);
		this.tokenizedLine.put("name", splittedLine[1]);
		this.tokenizedLine.put("longitude", splittedLine[2]);
		this.tokenizedLine.put("latitude", splittedLine[3]);
		this.tokenizedLine.put("height", splittedLine[4]);
	}

	public int getSimulationTime() {
		return (Integer.parseInt(this.line));
	}

	public String getAircraftType() {
		return (this.tokenizedLine.get("type"));
	}

	public String getAircraftName() {
		return (this.tokenizedLine.get("name"));
	}

	public Coordinates getAircraftCoordinates() {
		return (new Coordinates(
				Integer.parseInt(this.tokenizedLine.get("longitude")),
				Integer.parseInt(this.tokenizedLine.get("latitude")),
				Integer.parseInt(this.tokenizedLine.get("height"))
			)
		);
	}
}
